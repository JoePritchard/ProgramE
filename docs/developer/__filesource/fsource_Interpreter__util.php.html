  173. /**
    174. 

  174. * Substitution routine
    175. 

  175. *
    176. 

  176. * Is used in combination with for example {@link firstthird()} {@link gender()}. The myfunc() is called from these functions from the arrays generated in {@link subs.inc} by {@link makesrphp()}.
    177. 

  177. *
    178. 

  178. * When doing substitution myfunc replaces the words to be substituted with ~~x~~
    179. 

  179. * where x is an incremented integer instead of what should eventually be substituted.
    180. 

  180. * Then when all substitution is done another function will go through and replace the
    181. 

  181. * ~~x~~ with the real value.
    182. 

  182. *
    183. 

  183. * @todo Analyse if a straight replace will be more effective and efficient. Hard to see why this two-stage replace has any advantages. If not, then rename the function.
    184. 

  184. *
    185. 

  185. * @global integer replacecounter  The incremented integer number for substitution.
    186. 

  186. * @global array aftersearch       Contains ~~replacecounter number~~
    187. 

  187. * @global array afterreplace      Countains the input. (Anne:Have no idea how this works, yet.)
    188. 

  188. *
    189. 

  189. * @param string $input    The new word replacing the old word.
    190. 

  190. *
    191. 

  191. * @return string          The user's input, with ~~number~~ where the words in the substitution list
    192. 

  192. *                          should be.
    193. 

  193. */
    194. 

  194. function myfunc($input){
    195. 

  195.     
    196. 

  196.     global $replacecounter,$aftersearch,$afterreplace;
    197. 

  197.  
    198. 

  198.     $aftersearch[]="~~" . $replacecounter . "~~";
    199. 

  199.     $afterreplace[]=$input;
    200. 

  200.  
    201. 

  201.     return "~~" . $replacecounter++ . "~~";
    202. 

  202.  
    203. 

  203. }
