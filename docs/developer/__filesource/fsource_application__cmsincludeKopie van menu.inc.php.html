  110. function PrintNavigation($pat_id, $pid, $botid) {

  113. global $engine;

  121.     $query = "select id, ifnull(word,if(ordera='1','_','*')) as word from patterns where parent = ".$pat_id." and bot = '".$botid."' order by word";
    122. 

  122.     //debugger($query,2);
    123. 

  123.     $selectcode = mysql_query($query);

  124.     if ($selectcode){

  125.         if(!mysql_numrows($selectcode)){
    126. 

  126.             return "NO results";

  127.         }

  128.     }

  129.   $NO_results = mysql_num_rows($selectcode);

  130.  
    131.

    132. 

  132.   // floor'ing the results of the outcome of pagenumber / 10
    133. 

  133.   // means that the page numbering startpoint.

  134.   // Example 1, pid = 6  -> 0.6 -> 0 --> 0
    135. 

  135.   // Example 2, pid = 56 -> 5.6 -> 5 --> 50
    136. 

  136.   $startpagenumber = floor($pid/$engine['no_resultpages'])*$engine['no_resultpages'];

  137.

  138.   // determining the end page number.

  139.   // If the startpagenumber + number of pageresults (10)

    141. 

  141.   // proceed with that otherwise take the needed number
    142. 

  142.   // of pages.

  143.   if(($startpagenumber + $engine['no_searchresults']) < (ceil($NO_results/$engine['no_searchresults']))) {
    144. 

  144.      $endpagenumber = $startpagenumber + $engine['no_resultpages'];
    145. 

  145.   } else {
    146. 

  146.      $endpagenumber = (ceil($NO_results/$engine['no_searchresults']));
    147. 

  147.   }

  148.

  151.     <td class=\"resultnumber\">&nbsp;</td>

  152.     <td>";

  153.   
    154. 

  154.   
    155. 

  155.   // unless it's the first page, add the "previous" link.

  158.   }

  159.

  160.  // reminence of old code. for compatibility reasons.

  162.

  163.  // the 1,2,3 pagenumbers.

  165.
