    27. 

  27. * Retrieves REQUEST variables (POST/GET) and turns them into regular variables

  33. * 
    34.

  37. function getRequest($param, $default)
    38. 

  38. {
    39. 

  39.     global $$param;
    40. 

  40.     if (!IsSet($$param))
    41. 

  41.     {
    42. 

  42.       if (!IsSet($_REQUEST[$param]))
    43. 

  43.         {
    44. 

  44.           $$param = $default;
    45. 

  45.         }
    46. 

  46.       else
    47. 

  47.         {
    48. 

  48.           $$param = $_REQUEST[$param];
    49. 

  49.         }

  50.     }
