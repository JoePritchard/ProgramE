  115. function reply($userinput,$uniqueid, $bot = 1){

  116.

  127.  
    128. 

  128.     $myresponse->errors="";
    129. 

  129.     
    130.

  131.     $selectbot=$bot;    
    132. 

  132.  
    133.

    146.

  170.     $myresponse->patternsmatched=$patternmatched;
    171. 

  171.     $myresponse->inputs=$inputmatched;
    172. 

  172.  
    173. 

  173.     ss_timing_stop("all");

  174.  
    175. 

  175.     $myresponse->timer=ss_timing_current("all");
    176. 

  176.

  177.     return $myresponse;
    178.

  200. function respond($sentence){
    201. 

  201.     
    202. 

  202.     global $that,$loopcounter,$patternmatched,$inputmatched;

  204.     $topic = bget("topic");

    208.

  211.     $inputstarvals=array();

  212.     $thatstarvals=array();

  213.     $topicstarvals=array();

  214.     
    215.

    216. 

  216.     flush();

  217.  
    218. 

  218.     if ($that==""){
    219. 

  219.         $that="<nothing>";
    220. 

  220.     }
    221. 

  221.     if ($topic==""){
    222. 

  222.         $topic="<nothing>";
    223. 

  223.     }
    224. 

  224.  
    225. 

  225.     
    226. 

  226.     if ($sentence==""){
    227. 

  227.         return "";
    228. 

  228.     }
    229. 

  229.     else{
    230. 

  230.  
    231. 

  231.         //If we found a template
    232. 

  232.         $template=gettemplate($sentence,$that,$topic,$inputstarvals,$thatstarvals,$topicstarvals,$s_patternmatched,$s_inputmatched);
    233. 

  233.  
    234. 

  234.         $patternmatched[]=$s_patternmatched;
    235. 

  235.         $inputmatched[]=$s_inputmatched;
    236. 

  236.             
    237. 

  237.         if ($template!=""){
    238. 

  238.  
    239. 

  239.             $template="<xml><TEMPLATE>" . $template . "</TEMPLATE></xml>";        
    240. 

  240.             debugger ("found template: $template",2);            
    241. 

  241.             
    242. 

  242.  
    243. 

  243.             $root=GetXMLTree($template);
    244. 

  244.  
    245. 

  245.  
    246. 

  246.             if     (!isset($root[0]['children'][0]['value'])){
    247. 

  247.                 $root=$root[0]['children'][0]['children'];
    248. 

  248.             }
    249. 

  249.             else {
    250. 

  250.                 $root=$root[0]['children'][0]['value'];
    251. 

  251.             }
    252. 

  252.  
    253. 

  253. /*
    254. 

  254.             if     ($root[0]['children'][0]['value']==""){
    255. 

  255.                 $root=$root[0]['children'][0]['children'];
    256. 

  256.             }
    257. 

  257.             else {
    258. 

  258.                 $root=$root[0]['children'][0]['value'];
    259. 

  259.             }
    260. 

  260. */
    261. 

  261.  
    262. 

  262.             $myresponse=recursechildren($root,$inputstarvals,$thatstarvals,$topicstarvals);
    263. 

  263.             debugger("recursechildren ret: $myresponse",3);
    264. 

  264.  
    265.

  266.  
    267.

  427.         }
    428. 

  428.         debugger("starindex: $starindex",3);
    429. 

  429.         //print_r($inputstar);
    430. 

  430.         return $inputstar[$starindex-1];
    431. 

  431.         
    432. 

  432.  
    433. 

  433.     }
    434. 

  434.     elseif (strtoupper($xmlnode["tag"])=="THATSTAR"){
    435. 

  435.  
    436. 

  436.         $mynode=upperkeysarray($xmlnode["attributes"]);
    437. 

  437.  
    438. 

  438.         //$starindex=$xmlnode["attributes"]["INDEX"];
    439. 

  439.         
    440. 

  440.         if (!((is_array($mynode))&&(isset($mynode["INDEX"])))){

    442.

    444.

    445. 

  445.         if ($starindex==""){
    446. 

  446.             $starindex="1";
    447. 

  447.         }
    448. 

  448.         debugger("starindex: $starindex",3);
    449. 

  449.         //print_r($inputstar);
    450. 

  450.         return $thatstar[$starindex-1];
    451. 

  451.         
    452. 

  452.  
    453. 

  453.     }
    454. 

  454.     elseif (strtoupper($xmlnode["tag"])=="TOPICSTAR"){
    455. 

  455.  
    456. 

  456.         
    457. 

  457.         $mynode=upperkeysarray($xmlnode["attributes"]);
    458. 

  458.  
    459. 

  459.         //$starindex=$xmlnode["attributes"]["INDEX"];
    460. 

  460.  
    461. 

  461.         if (!((is_array($mynode))&&(isset($mynode["INDEX"])))){
    462. 

  462.             $mynode["INDEX"]="";
    463. 

  463.         }
    464. 

  464.  
    465. 

  465.         $starindex=$mynode['INDEX'];
    466. 

  466.         if ($starindex==""){
    467. 

  467.             $starindex="1";
    468. 

  468.         }
    469. 

  469.         debugger("starindex: $starindex",3);
    470.

    471. 

  471.         return $topicstar[$starindex-1];
    472. 

  472.         
    473. 

  473.  
    474. 

  474.     }
    475. 

  475.     elseif (strtoupper($xmlnode["tag"])=="SRAI"){
    476. 

  476.         
    477. 

  477.         // Build up a new response inside of here (using recursechildren function and then call response with it.
    478.
